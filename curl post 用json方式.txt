curl post 用json方式
if(!function_exists('tps_curl_post3')){   function tps_curl_post3($url, $postData) {      $postData = json_encode($postData);      $curl = curl_init();  //初始化      curl_setopt($curl,CURLOPT_URL,$url);  //设置url      curl_setopt($curl,CURLOPT_HTTPAUTH,CURLAUTH_BASIC);  //设置http验证方法      curl_setopt($curl, CURLOPT_TIMEOUT,30);      curl_setopt($curl,CURLOPT_RETURNTRANSFER,1);  //设置curl_exec获取的信息的返回方式      curl_setopt($curl,CURLOPT_POST,1);  //设置发送方式为post请求      curl_setopt($curl,CURLOPT_POSTFIELDS,$postData);  //设置post的数据      curl_setopt($curl, CURLOPT_HTTPHEADER, array(            'Content-Type: application/json',            'Content-Length: ' . strlen($postData))      );      $result = curl_exec($curl);      curl_close($curl);      return json_decode($result,true);   }}
